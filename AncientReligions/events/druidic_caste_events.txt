character_event = {
	id = ancrel.0186
	hide_window = yes
	
	ai = yes
	min_age = 16
	max_age = 25
	only_men = yes
	capable_only = yes
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		is_landed = no
		in_command = no
		OR = {
			religion = druidic
			religion = druidic_reformed
		}
		NOT = {
			OR = {
				trait = cynical
				trait = imbecile
				trait = slow
				trait = craven
			}
			has_character_flag = wants_druidic_training
		}
		father = {
			is_alive = yes
			OR = {
				religion = druidic
				religion = druidic_reformed
			}
			is_ruler = yes
			is_liege_of = ROOT
			prisoner = no
			NOT = { trait = incapable }
			ROOT = { host = { character = PREVPREV } }
		}
	}
	
	mean_time_to_happen = {
		months = 450
		
		modifier = {
			factor = 0.5
			trait = content
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.75
			trait = quick
		}
		modifier = {
			factor = 0.75
			trait = genius
		}
		modifier = {
			factor = 1.5
			trait = c
		}
		modifier = {
			factor = 1.5
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			is_primary_heir = yes
		}
		modifier = {
			factor = 2.0
			trait = hunter
		}
	}
	
	immediate = {
		father = { character_event = { id = ancrel.0187 } }
		set_character_flag = wants_druidic_training
	}
	
	option = {
		name = OK
	}
}
