	OR = {
		religion = celtic_pagan_reformed
		religion = celtic_pagan
		religion = promethean_pagan
		religion = messalian
		religion = cathar
		primary_title = { has_law = status_of_women_4 }
	}
}

is_attracted_to_prev = {
	is_valid_attraction = PREV
	OR = {
		AND = {
			is_female = no
			OR = {
				AND = {
					trait = homosexual
					PREV = { is_female = no }
				}
				PREV = { is_female = yes }
			}
		}
		AND = {
			is_female = yes
			OR = {
				AND = {
					trait = homosexual
					PREV = { is_female = yes }
				}
				PREV = { is_female = no }
			}
		}
	}

	opinion = { who = PREV value = 30 }
}
